1. Exception: Undeclared Variable

try {
console.log(myVar); // myVar is not declared
} catch (error) {
console.log("Error caught: " + error.message);
}

2. Throw Divide by Zero Error

function divide(a, b) {
if (b === 0) {
throw new Error("Divide by zero error!");
}
return a / b;
}

try {
console.log(divide(10, 0));
} catch (error) {
console.log("Error caught: " + error.message);
}

3. Throw Out of Bound Error

function getElement(arr, index) {
if (index >= arr.length || index < 0) {
throw new Error("Out of bounds error!");
}
return arr[index];
}

try {
let numbers = [1, 2, 3];
console.log(getElement(numbers, 5));
} catch (error) {
console.log("Error caught: " + error.message);
}

4. Exception Handling with Finally

try {
let result = 10 / 0;
if (!isFinite(result)) {
throw new Error("Division by zero");
}
console.log("Result:", result);
} catch (error) {
console.log("Caught error:", error.message);
} finally {
console.log("This block always executes (finally).");
}
